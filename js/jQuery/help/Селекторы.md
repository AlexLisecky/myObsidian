
#### Здесь собраны селекторы jQuery

```js
// Выбор всех элементов
$('*')
```
***
```js
// Выбор элемента по ID
$('#id')
```
***
```js
// Выбор HTML элемента
$('form') $('button') $('body') $('div')
```
***
```js
// Множественный выбор элементов 
$('form', '#id', '.class', 'a')
```
***
```js
// Выбор непосредственный дочерних элементов
$('div > a') // выберет все ссылки у которых родитель div
$('.class > .newclass')
```
***
```js
// выбор элементов которые расположены сразу после элемента
$('h2 + p') // выберет все элементы после h2
$('button + .class')
```
***
```js
// выбор сестинцких элементов
$('div ~ p') // выберет все элементы p которые находстя на одном уровне с div
```
***
```js
$('button ~ .class') // выберет элементы c классом .class которые находсся на одном уровне с button
```
***
```js
// выбор первого элемента, к выборке добавляем :first
$('a:first') // выберет первый элемент a
$('button:first') // выберет первый элемент button
$('.class:first') // выберет первый элемент c классом .class
```
***
```js
// выбор последнего элемента, к выборке добавляем :last
$('a:last') // выберет последний элемент a
$('button:last') // выберет последний элемент button
$('.class:last') // выберет последний элемент c классом .class
```
***
```js
// выбор четный элементов
$('.class:even')
$('button:even')
```
***
```js
// выбор нечетных элементов
$('.class:odd')
$('.class:odd')
```
