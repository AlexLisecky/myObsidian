### Описание ***методов*** для взаимодействия с элементами

```js
// .addClass() 
// Добавляет один или несколько имен классов для выбранных элементов.
$('.custom').addClass('active');
// может добавить несколько классов
$('.custom').addClass('active new_class');
```
***
```js
// .addClass() 
// Добавляет один или несколько имен классов для выбранных элементов.
$('.custom').removeClass('active');
// может добавить несколько классов
$('.custom').removeClass('active new_class');
```
***
```js
// .after()
// Вставляет указанное содержимое, переданное в параметре метода после каждого выбранного элемента.
$('.custom').after('<b>Hello</b>');
// может вставить взятый элемент(элемент не клонируется а перемещается)
$('.custom').after($('.block'))
```
***
```js
// .before()
// Вставляет указанное содержимое, переданное в параметре метода перед каждыми выбранным элементом.
$('.custom').before('<b>Hello</b>');
// может вставить взятый элемент(элемент не клонируется а перемещается)
$('.custom').before($('.block'))
```
***
```js
// .append()
// Вставляет содержимое, указанное в параметре метода в конец каждого выбранного элемента в наборе совпавших элементов.
// ВАЖНО: вставляет именно в элемент
$('.custom').append('<b>Hello</b>')
// может вставить взятый элемент(элемент не клонируется а перемещается)
$('.custom').append($('.block'))
```
***
```js
// .prepend()
// Вставляет содержимое, указанное в параметре метода в начало каждого выбранного элемента в наборе совпавших элементов.
// ВАЖНО: вставляет именно в элемент
$('.custom').prepend('<b>Hello</b>')
// может вставить взятый элемент(элемент не клонируется а перемещается)
$('.custom').prepend($('.block'))
```
***
```js
// attr()
// Получает значение атрибута для первого элемента в наборе совпавших элементов, или устанавливает один или несколько атрибутов для набора выбранных элементов.
<div title='Hello' data-status='ok' class="custom"></div>
$('.custom').attr('title') --> 'Hello'
$('.custom').attr('data-status') --> 'ok'

// если будет два элемента возьмет только ПЕРВЫЙ
<div title='Hello' data-status='ok' class="custom"></div>
<div title='не возьмет' data-status='не возьмет' class="custom"></div>
$('.custom').attr('title') --> 'Hello'
$('.custom').attr('data-status') --> 'ok'

// установить значение атрибута
$('.custom').attr('title', 'Новый тайтл')
// множественная установка
$('.custom').attr({'title': 'Новый тайтд', 'class': 'Новый класс'})

// убираем с инпутов атрибуты
$('input').attr('required', false)
$('input').attr('checked', false)

// добавляем к инпутам атрибуты
$('input').attr('required', true)
$('input').attr('checked', true)

// добавляем к видео элементы контроля
$('video').attr('controls', true)
```
***
```js
// .clone()
// Создает копию выбранных элементов, включая дочерние узлы, текст и атрибуты.
let elem = $('.custom').clone();
$('.custom').append(elem);
$('.custom').after(elem);
```
